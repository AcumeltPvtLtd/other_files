        (function() {
            // --- Setup scene, camera, renderers (WebGL + CSS2D) ---
            const scene = new THREE.Scene();
            scene.background = new THREE.Color(0x111111);

            const camera = new THREE.PerspectiveCamera(45, window.innerWidth / window.innerHeight, 1, 5000);
            camera.position.set(1000, 600, 1000);
            camera.lookAt(0, 250, 0);

            // WebGL renderer
            const renderer = new THREE.WebGLRenderer({ antialias: true });
            renderer.setSize(window.innerWidth, window.innerHeight);
            renderer.setPixelRatio(window.devicePixelRatio);
            renderer.shadowMap.enabled = false;
            document.body.appendChild(renderer.domElement);

            // CSS2 renderer for labels
            const labelRenderer = new THREE.CSS2DRenderer();
            labelRenderer.setSize(window.innerWidth, window.innerHeight);
            labelRenderer.domElement.style.position = 'absolute';
            labelRenderer.domElement.style.top = '0px';
            labelRenderer.domElement.style.left = '0px';
            labelRenderer.domElement.style.pointerEvents = 'none';
            labelRenderer.domElement.classList.add('css2d-label');
            document.body.appendChild(labelRenderer.domElement);

            // --- Controls ---
            const controls = new THREE.OrbitControls(camera, renderer.domElement);
            controls.target.set(0, 250, 0);
            controls.enableDamping = true;
            controls.dampingFactor = 0.05;
            controls.screenSpacePanning = true;
            controls.maxPolarAngle = Math.PI / 2;

            // --- Lighting (increased intensity for better contrast) ---
            const ambientLight = new THREE.AmbientLight(0x606060);
            scene.add(ambientLight);

            const dirLight = new THREE.DirectionalLight(0xffffff, 1.2);
            dirLight.position.set(500, 1000, 500);
            scene.add(dirLight);

            const fillLight = new THREE.DirectionalLight(0xccddff, 0.6);
            fillLight.position.set(-300, 400, -400);
            scene.add(fillLight);

            // --- Visual Aids: Ground grid only ---
            const gridHelper = new THREE.GridHelper(1200, 12, 0x888888, 0x444444);
            gridHelper.position.y = 0;
            scene.add(gridHelper);

            // ========== 1. MOLD OUTER BOX ==========
            const outerW = 508;  // X
            const outerD = 660;  // Z
            const outerH = 500;  // Y

            // Semi-transparent grey faces (opacity 0.2 for better visibility)
            const outerGeo = new THREE.BoxGeometry(outerW, outerH, outerD);
            const outerMat = new THREE.MeshPhongMaterial({
                color: 0xaaaaaa,
                transparent: true,
                opacity: 0.2,
                side: THREE.DoubleSide
            });
            const outerMesh = new THREE.Mesh(outerGeo, outerMat);
            outerMesh.position.set(0, outerH / 2, 0);
            scene.add(outerMesh);

            // Black edges
            const outerEdges = new THREE.EdgesGeometry(outerGeo);
            const outerLines = new THREE.LineSegments(outerEdges, new THREE.LineBasicMaterial({ color: 0x000000 }));
            outerLines.position.copy(outerMesh.position);
            scene.add(outerLines);

            // ========== 2. CAVITY (max component volume) ==========
            const offsetSide = 50;
            const offsetBottom = 70;
            const offsetTop = 90;

            const cavityW = outerW - 2 * offsetSide;   // 408
            const cavityD = outerD - 2 * offsetSide;   // 560
            const cavityH = outerH - (offsetBottom + offsetTop); // 340
            const cavityCenterY = offsetBottom + cavityH / 2; // 240

            // Light grey edges for cavity
            const cavityGeo = new THREE.BoxGeometry(cavityW, cavityH, cavityD);
            const cavityEdges = new THREE.EdgesGeometry(cavityGeo);
            const cavityLines = new THREE.LineSegments(cavityEdges, new THREE.LineBasicMaterial({ color: 0xcccccc }));
            cavityLines.position.set(0, cavityCenterY, 0);
            scene.add(cavityLines);

            // ========== 3. PARTING LINE (Y = 250) – 60% TRANSPARENT ORANGE PLANE ==========
            const partingPlaneGeo = new THREE.PlaneGeometry(outerW, outerD);
            const partingPlaneMat = new THREE.MeshPhongMaterial({
                color: 0xff6600,
                transparent: true,
                opacity: 0.4,        // 60% transparent (40% opaque)
                side: THREE.DoubleSide
            });
            const partingPlane = new THREE.Mesh(partingPlaneGeo, partingPlaneMat);
            partingPlane.rotation.x = -Math.PI / 2;
            partingPlane.position.set(0, 250, 0);
            scene.add(partingPlane);

            // Orange perimeter line for extra definition
            const halfW = outerW / 2;
            const halfD = outerD / 2;
            const corners = [
                new THREE.Vector3( halfW, 250,  halfD),
                new THREE.Vector3(-halfW, 250,  halfD),
                new THREE.Vector3(-halfW, 250, -halfD),
                new THREE.Vector3( halfW, 250, -halfD)
            ];
            const loopPoints = [corners[0], corners[1], corners[2], corners[3], corners[0]];
            const loopGeo = new THREE.BufferGeometry().setFromPoints(loopPoints);
            const loopMat = new THREE.LineBasicMaterial({ color: 0xff6600 });
            const partingLoop = new THREE.Line(loopGeo, loopMat);
            scene.add(partingLoop);

            // ========== 4. SPRUE CUPS (top face, Y=500) ==========
            const cupRadius = 25;   // diameter 50
            const cupHeight = 1;
            const cupGeo = new THREE.CylinderGeometry(cupRadius, cupRadius, cupHeight, 32);
            const cupMat = new THREE.MeshStandardMaterial({ color: 0xff6600, emissive: 0x221100 });
            const cupY = 500 + cupHeight / 2;

            const cupPositions = [
                { x: 0, z: 0 },
                { x: 160, z: 0 },
                { x: 0, z: 235 },
                { x: 160, z: 235 }
            ];

            cupPositions.forEach(pos => {
                const cup = new THREE.Mesh(cupGeo, cupMat);
                cup.position.set(pos.x, cupY, pos.z);
                scene.add(cup);
            });

            // ========== 5. DIMENSION LABELS (CSS2D) – SINGLE PER AXIS, NO REPETITION ==========
            function createLabel(text) {
                const div = document.createElement('div');
                div.textContent = text;
                div.style.color = '#ffffff';
                div.style.fontSize = '20px';
                div.style.fontWeight = 'bold';
                div.style.fontFamily = '"Courier New", Courier, monospace';

                div.style.borderRadius = '4px';
                div.style.border = '1px solid #333';
                return new THREE.CSS2DObject(div);
            }

            // - Front face (Z = +280) : depth 560 mm
            // - Back face (Z = -280) : height 340 mm

            const labelWidth = createLabel('408 mm');   // width (X)
            labelWidth.position.set(204, 240, 0);
            scene.add(labelWidth);

            const labelDepth = createLabel('560 mm');   // depth (Z)
            labelDepth.position.set(0, 240, 280);
            scene.add(labelDepth);

            const labelHeight = createLabel('340 mm');  // height (Y)
            labelHeight.position.set(0, 240, -280);
            scene.add(labelHeight);

            // (No outer mold labels, no central cavity label – only these three axis labels)

            // ========== ANIMATION LOOP ==========
            function animate() {

            animate();

            // ========== RESIZE HANDLER ==========
            window.addEventListener('resize', onWindowResize, false);
            function onWindowResize() {
                camera.aspect = window.innerWidth / window.innerHeight;
                camera.updateProjectionMatrix();
                renderer.setSize(window.innerWidth, window.innerHeight);
                labelRenderer.setSize(window.innerWidth, window.innerHeight);
            }

        })();
